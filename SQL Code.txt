4.  	Function : การชำระเงิน

SELECT od.MenuID, od.Quantity, od.UnitPrice,
(od.Quantity * od.UnitPrice) AS Subtotal,  
SUM(od.Quantity * od.UnitPrice) OVER() AS P_total 
FROM OrderDetails od
WHERE od.OId = '<รหัสออเดอร์>';
 
เป็นฟังก์ชั่นสำหรับการคำนวณราคารวมในแต่ละรายการอาหาร และการคำนวณยอดรวมทั้งหมดที่ต้องชำระเงิน
 
INSERT INTO Payment (PId, P_Method, P_DateTime, OId, P_total )
VALUES ('<รหัสการชำระเงิน>','<รูปแบบการชำระเงิน>', NOW( ), '<รหัสรายการสั่งซื้อ>', (SELECT SUM(od.Quantity * od.UnitPrice) FROM OrderDetails  WHERE OId = '<รหัสออเดอร์>') );
 
บันทึกรหัสการชำระเงิน รูปแบบการชำระเงิน วันและเวลาที่ทำการชำระเงิน รหัสรายการสั่งซื้อและยอดรวมลงในตารางPayment

11. 	 Function : การค้นหาและแสดงรายการอาหาร

SELECT m.MenuId, m.MenuName, m.Price, c.Category_Name
FROM Menu m
JOIN Categorizers cg ON m.MenuId = cg.MenuId
JOIN Category c ON cg.Category_Id = c.Category_Id
WHERE m.MenuName LIKE '%<คำค้านหา>%'
OR c.Category_Name = '<คำค้นหา>';

เป็นฟังก์ชันค้นหารายการอาหารจากชื่อรายการอาหารหรือชื่อหมวดหมู่จากฐานข้อมูล และแสดงรายการอาหาร
